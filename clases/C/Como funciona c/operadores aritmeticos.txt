Suma

int main()
{
    int num1;
    int num2;
    int resultado;
    printf("ingrese el primer valor");
    scanf("%d",&num1);
    printf("ingrese el segundo valor");
    scanf("%d",&num2);
    resultado = num1 + num2;
    printf("el resultado de la suma es:%d\n", resultado);
    return 0;
}
